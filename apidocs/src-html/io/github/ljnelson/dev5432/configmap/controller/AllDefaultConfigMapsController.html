064/**
065 * A <em>controller</em> that {@linkplain
066 * #onConfigMapAddition(ConfigMap) watches for the addition of
067 * <code>ConfigMap</code>}s.
068 *
069 * @author <a href="https://about.me/lairdnelson"
070 * target="_parent">Laird Nelson</a>
071 *
072 * @see #onConfigMapAddition(ConfigMap)
073 *
074 * @see #produceAllDefaultConfigMaps(KubernetesClient)
075 */
076@ApplicationScoped
077final class AllDefaultConfigMapsController {
078
079  /**
080   * A {@link Logger} for logging messages.
081   *
082   * <p>This field is never {@code null}.</p>
083   *
084   * <p>This {@link Logger}'s {@linkplain Logger#getName() name} is
085   * {@code
086   * ljnelson.dev5432.configmap.controller.AllDefaultConfigMapsController}.</p>
087   */
088  private static final Logger logger = Logger.getLogger(AllDefaultConfigMapsController.class.getName());
089
090  /**
091   * Creates a new {@link AllDefaultConfigMapsController}.
092   */
093  private AllDefaultConfigMapsController() {
094    super();
095  }
096
097  /**
098   * A <a
099   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#producer_method"
100   * target="_parent">producer method</a> which is a <em>Kubernetes
101   * event selector</em> that produces a {@link NonNamespaceOperation}
102   * whose {@link Listable#list() list()} and {@link
103   * Watchable#watch(Object) watch(Watcher)} methods will eventually
104   * produce events that will be processed by the {@link
105   * #onConfigMapAddition(ConfigMap)} Kubernetes event observer
106   * method.
107   *
108   * <p>This method never returns {@code null}.</p>
109   *
110   * @param client a {@link KubernetesClient} injected by the CDI ecosystem;
111   * will not be {@code null}; in this example application it is
112   * supplied to CDI by the <a
113   * href="https://microbean.github.io/microbean-kubernetes-client-cdi/"
114   * target="_parent">microBean Kubernetes Client CDI</a> project
115   *
116   * @return the result of invoking {@link
117   * Namespaceable#inNamespace(String)
118   * client.configMaps().inNamespace("default")}; never {@code null}
119   *
120   * @see KubernetesClient
121   *
122   * @see KubernetesClient#configMaps()
123   *
124   * @see Namespaceable#inNamespace(String)
125   *
126   * @see #onConfigMapAddition(ConfigMap)
127   */
128  @Produces
129  @ApplicationScoped
130  @AllDefaultConfigMaps
131  private static final NonNamespaceOperation<ConfigMap, ConfigMapList, DoneableConfigMap, Resource<ConfigMap, DoneableConfigMap>> produceAllDefaultConfigMaps(final KubernetesClient client) {
132    return client.configMaps().inNamespace("default");
133  }
134
135  /**
136   * A <em>Kubernetes event observer method</em>
137   * that will be notified with {@link ConfigMap} "events" that
138   * describe the {@linkplain Added addition} of a {@code ConfigMap}
139   * to a Kubernetes cluster.
140   *
141   * <p>The production of these events is governed and filtered by (in
142   * this application) the recipe manifested by the {@link
143   * #produceAllDefaultConfigMaps(KubernetesClient)} producer method.
144   * The events that result indirectly from that Kubernetes event
145   * selector method are "linked" with this Kubernetes event observer
146   * method by way of the {@link AllDefaultConfigMaps} Kubernetes
147   * event selector annotation.</p>
148   *
149   * @param newConfigMap the {@link ConfigMap} that was added; will
150   * not be {@code null}
151   *
152   * @see AllDefaultConfigMaps
153   *
154   * @see #produceAllDefaultConfigMaps(KubernetesClient)
155   *
156   * @see Added
157   */
158  private final void onConfigMapAddition(@Observes
159                                         @AllDefaultConfigMaps
160                                         @Added
161                                         final ConfigMap newConfigMap) {
162    logger.log(Level.INFO, "New ConfigMap added: {0}", newConfigMap);
163  }
164
165  private final void onConfigMapModification(@Observes
166                                             @AllDefaultConfigMaps
167                                             @Modified
168                                             final ConfigMap modifiedConfigMap,
169                                             @Prior
170                                             final Optional<ConfigMap> priorConfigMap) {
171    logger.log(Level.INFO, "ConfigMap modified.\n  Old ConfigMap:\n{0}\n  New ConfigMap:\n{1}\n", new Object[] { priorConfigMap.get(), modifiedConfigMap });
172  }
173
174  /**
175   * A <a
176   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#producer_method"
177   * target="_parent">producer method</a> that returns an {@linkplain
178   * ApplicationScoped application-scoped} {@link Map} of {@link
179   * ConfigMap}s indexed by the keys that Kubernetes assigns to them.
180   *
181   * <p>This method never returns {@code null}.</p>
182   *
183   * <h2>Thread Safety</h2>
184   *
185   * <p>This method is safe for concurrent use by multiple
186   * threads.</p>
187   *
188   * <p><strong>The return value that results from invoking this
189   * method is <em>not</em> safe for concurrent use by multiple
190   * threads.</strong> Threads must {@code synchronize} on it
191   * appropriately for all retrieval, traversal and mutation
192   * operations.</p>
193   *
194   * @return a non-{@code null} {@link Map} that will be used to house
195   * prior states of Kubernetes resources
196   */
197  @Produces
198  @ApplicationScoped
199  @AllDefaultConfigMaps
200  private static final Map<Object, ConfigMap> produceKubernetesResourceCache() {
201    return new HashMap<>();
202  }
203  
204  /**
205   * A {@linkplain Qualifier qualifier annotation} that itself is
206   * qualified with {@link KubernetesEventSelector} allowing it to pair
207   * relevant producer and observer methods so that certain {@code
208   * ConfigMap} resources can be observed.
209   *
210   * <h2>Learning Notes</h2>
211   *
212   * <p>Note that this annotation is annotated with both {@link
213   * Qualifier}&mdash;making it a <a
214   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#qualifiers"
215   * target="_parent">CDI qualifier</a>&mdash;and {@link
216   * KubernetesEventSelector}&mdash;allowing it to be recognized by the
217   * <a
218   * href="https://microbean.github.io/microbean-kubernetes-controller-cdi/"
219   * target="_parent">microBean Kubernetes Controller CDI</a>
220   * framework.</p>
221   *
222   * <p>Because it "is a" {@link KubernetesEventSelector}, you apply
223   * this annotation to two things:</p>
224   *
225   * <ol>
226   *
227   * <li>To a bean&mdash;most commonly to a <a
228   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#producer_method"
229   * target="_parent">producer method</a>&mdash;that can be assigned to
230   * a reference that can be described by the following type
231   * declaration: {@code <X extends }{@link Listable}{@code <?  extends
232   * }{@link KubernetesResourceList}{@code > & }{@link
233   * VersionWatchable}{@code <? extends }{@link Closeable}{@code ,
234   * }{@link Watcher}{@code <? extends }{@link HasMetadata}{@code >>>}.
235   * This bean is known as a <em>Kubernetes event selector</em>: it is
236   * something whose {@link Listable#list() list()} method and {@link
237   * Watchable#watch(Object) watch(W)} method will be called to select
238   * what Kubernetes events will get delivered to your Kubernetes event
239   * observer method.  While that type declaration looks complicated, it
240   * conveniently describes the return types of most of the methods on
241   * {@link KubernetesClient}.</li>
242   *
243   * <li>To an <a
244   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#observer_methods"
245   * target="_parent">observer method</a>'s <a
246   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#observer_method_event_parameter"
247   * target="_parent">event parameter</a> whose type is identical to the
248   * "primary type" in the matching bean or producer method.  For
249   * example, given a bean type&mdash;usually the return value of a
250   * producer method&mdash;of {@link NonNamespaceOperation}{@code
251   * <}{@link ConfigMap}{@code , }{@link ConfigMapList}{@code , }{@link
252   * DoneableConfigMap}{@code , }{@link Resource}{@code <}{@link
253   * ConfigMap}{@code , }{@link DoneableConfigMap}{@code >>} the
254   * "primary type" is {@link ConfigMap}.  This identifies your observer
255   * method to the <a
256   * href="https://microbean.github.io/microbean-kubernetes-controller-cdi/"
257   * target="_parent">microBean Kubernetes Controller CDI</a> framework
258   * and allows it to deliver Kubernetes events to it using the
259   * similarly-annotated bean or producer method (described above).  An
260   * observer method whose event parameter is annotated in this
261   * way&mdash;together with {@link Added}, {@link Modified} or {@link
262   * Deleted} qualifier annotations&mdash;constitutes a <em>Kubernetes
263   * event observer method</em>.</li>
264   *
265   * </ol>
266   *
267   * <p>When you "connect" a Kubernetes event selector with a Kubernetes
268   * event observer method using this annotation, then events produced
269   * by the Kubernetes event selector will be delivered to the
270   * Kubernetes event observer method.  If you "connect" a different
271   * Kubernetes event selector with a different Kubernetes event
272   * observer method using another {@link
273   * KubernetesEventSelector}-annotated qualifier annotation, then
274   * <em>that</em> Kubernetes event selector will be used to deliver
275   * Kubernetes events to <em>that</em> Kubernetes event observer
276   * method.</p>
277   *
278   * @see KubernetesEventSelector
279   */
280  @Documented
281  @KubernetesEventSelector
282  @Qualifier
283  @Retention(value = RetentionPolicy.RUNTIME)
284  @Target({ ElementType.FIELD, ElementType.METHOD, ElementType.PARAMETER, ElementType.TYPE })
285  public @interface AllDefaultConfigMaps {
286
287  }
